```python
from google.adk.tools import VertexAiRagRetrieval

destination_knowledge = VertexAiRagRetrieval(
    name='destination_knowledge',
    description='''Retrieve destination information from travel guide knowledge base.

    USE THIS TOOL to answer questions about:
    - Visa requirements and entry rules
    - Top attractions and landmarks
    [...]

    DO NOT use this tool for:
    - Real-time flight or hotel availability
    [...]''',
    rag_corpus_resource=os.environ.get('RAG_CORPUS_ID'),
    similarity_top_k=5,
)

# Why This Works:
# - DO/DO NOT sections prevent LLM from misusing the tool
# - similarity_top_k controls how many chunks are retrieved
# - Explicit description boundaries ensure correct tool selection

# Key Insight: RAG tool descriptions need explicit boundaries.
# Without DO/DO NOT sections, the LLM may call RAG for real-time
# queries that tools should handle instead.
```
