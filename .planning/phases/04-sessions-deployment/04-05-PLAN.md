"""
Cost Tracking Utility for Workshop Usage

Tracks token usage and API costs for the travel booking assistant.
Useful for monitoring workshop expenses and optimizing agent prompts.

Pricing (Gemini 2.5 Flash as of 2026):
- Input: $0.30 per 1M tokens
- Output: $2.50 per 1M tokens

Usage:
    from cost_tracker import WorkshopCostTracker

    tracker = WorkshopCostTracker()
    # After each agent query
    tracker.log_query(user_id, query, response)
    # Get summary
    print(tracker.get_summary())

Exercise 4 Reference - Cost Monitoring (INFRA-08)
"""

from dataclasses import dataclass, field
from typing import List, Dict, Optional
from datetime import datetime
import json


# ============================================================
# PRICING CONFIGURATION
# ============================================================

# Gemini 2.5 Flash pricing (USD per 1M tokens)
# Source: https://ai.google.dev/gemini-api/docs/pricing
PRICING = {
    "gemini-2.5-flash": {
        "input_per_1m": 0.30,
        "output_per_1m": 2.50,
    },
    "gemini-2.0-flash": {
        "input_per_1m": 0.10,
        "output_per_1m": 0.40,
    },
}

DEFAULT_MODEL = "gemini-2.5-flash"


# ============================================================
# DATA CLASSES
# ============================================================

@dataclass
class TokenUsage:
    """Single query token usage record."""
    user_id: str
    timestamp: datetime
    query: str
    input_tokens: int
    output_tokens: int
    cost_usd: float
    model: str = DEFAULT_MODEL


@dataclass
class CostSummary:
    """Aggregated cost summary."""
    total_queries: int
    unique_users: int
    total_input_tokens: int
    total_output_tokens: int
    total_cost_usd: float
    avg_cost_per_query: float
    avg_tokens_per_query: float
    model: str


# ============================================================
# COST TRACKER
# ============================================================

class WorkshopCostTracker:
    """
    Track token usage and costs during workshop sessions.

    Example:
        tracker = WorkshopCostTracker()

        # After each agent response
        tracker.log_query(
            user_id="participant_1",
            query="Find flights to Tokyo",
            response=agent_response  # Has usage_metadata
        )

        # Get workshop totals
        summary = tracker.get_summary()
        print(f"Workshop cost: ${summary.total_cost_usd:.2f}")
    """

    def __init__(self, model: str = DEFAULT_MODEL):
        """Initialize tracker with pricing model."""
        self.model = model
        self.usage_log: List[TokenUsage] = []

        if model not in PRICING:
            raise ValueError(f"Unknown model: {model}. Known: {list(PRICING.keys())}")

        self.input_price = PRICING[model]["input_per_1m"]
        self.output_price = PRICING[model]["output_per_1m"]

    def log_query(
        self,
        user_id: str,
        query: str,
        response,  # GenAI response with usage_metadata
    ) -> TokenUsage:
        """
        Log token usage from an agent response.

        Args:
            user_id: Participant identifier
            query: User's query text
            response: Agent response with usage_metadata attribute

        Returns:
            TokenUsage record
        """
        # Extract token counts from response
        if hasattr(response, 'usage_metadata'):
            metadata = response.usage_metadata
            input_tokens = getattr(metadata, 'prompt_tokens', 0)
            output_tokens = getattr(metadata, 'candidates_tokens', 0)
        else:
            # Estimate if metadata not available
            input_tokens = len(query.split()) * 2  # Rough estimate
            output_tokens = 100  # Default estimate
            print(f"Warning: No usage_metadata, using estimates")

        # Calculate cost
        input_cost = (input_tokens / 1_000_000) * self.input_price
        output_cost = (output_tokens / 1_000_000) * self.output_price
        total_cost = input_cost + output_cost

        # Create record
        usage = TokenUsage(
            user_id=user_id,
            timestamp=datetime.now(),
            query=query[:100],  # Truncate for storage
            input_tokens=input_tokens,
            output_tokens=output_tokens,
            cost_usd=total_cost,
            model=self.model,
        )

        self.usage_log.append(usage)
        return usage

    def log_tokens_directly(
        self,
        user_id: str,
        query: str,
        input_tokens: int,
        output_tokens: int,
    ) -> TokenUsage:
        """
        Log token counts directly (for testing or manual tracking).

        Args:
            user_id: Participant identifier
            query: User's query text
            input_tokens: Number of input tokens
            output_tokens: Number of output tokens

        Returns:
            TokenUsage record
        """
        input_cost = (input_tokens / 1_000_000) * self.input_price
        output_cost = (output_tokens / 1_000_000) * self.output_price
        total_cost = input_cost + output_cost

        usage = TokenUsage(
            user_id=user_id,
            timestamp=datetime.now(),
            query=query[:100],
            input_tokens=input_tokens,
            output_tokens=output_tokens,
            cost_usd=total_cost,
            model=self.model,
        )

        self.usage_log.append(usage)
        return usage

    def get_summary(self) -> CostSummary:
        """
        Get aggregated cost summary.

        Returns:
            CostSummary with totals and averages
        """
        if not self.usage_log:
            return CostSummary(
                total_queries=0,
                unique_users=0,
                total_input_tokens=0,
                total_output_tokens=0,
                total_cost_usd=0.0,
                avg_cost_per_query=0.0,
                avg_tokens_per_query=0.0,
                model=self.model,
            )

        total_input = sum(u.input_tokens for u in self.usage_log)
        total_output = sum(u.output_tokens for u in self.usage_log)
        total_cost = sum(u.cost_usd for u in self.usage_log)
        unique_users = len(set(u.user_id for u in self.usage_log))
        total_queries = len(self.usage_log)

        return CostSummary(
            total_queries=total_queries,
            unique_users=unique_users,
            total_input_tokens=total_input,
            total_output_tokens=total_output,
            total_cost_usd=total_cost,
            avg_cost_per_query=total_cost / total_queries,
            avg_tokens_per_query=(total_input + total_output) / total_queries,
            model=self.model,
        )

    def get_user_summary(self, user_id: str) -> Dict:
        """Get cost summary for a specific user."""
        user_logs = [u for u in self.usage_log if u.user_id == user_id]
        if not user_logs:
            return {"user_id": user_id, "queries": 0, "cost_usd": 0.0}

        return {
            "user_id": user_id,
            "queries": len(user_logs),
            "input_tokens": sum(u.input_tokens for u in user_logs),
            "output_tokens": sum(u.output_tokens for u in user_logs),
            "cost_usd": sum(u.cost_usd for u in user_logs),
        }

    def print_report(self):
        """Print formatted cost report."""
        summary = self.get_summary()

        print("=" * 60)
        print("WORKSHOP COST REPORT")
        print("=" * 60)
        print(f"Model: {summary.model}")
        print(f"Total queries: {summary.total_queries}")
        print(f"Unique users: {summary.unique_users}")
        print("-" * 60)
        print(f"Input tokens: {summary.total_input_tokens:,}")
        print(f"Output tokens: {summary.total_output_tokens:,}")
        print(f"Total tokens: {summary.total_input_tokens + summary.total_output_tokens:,}")
        print("-" * 60)
        print(f"Total cost: ${summary.total_cost_usd:.4f}")
        print(f"Avg per query: ${summary.avg_cost_per_query:.4f}")
        print(f"Avg tokens/query: {summary.avg_tokens_per_query:.0f}")
        print("=" * 60)

    def export_json(self, filepath: str):
        """Export usage log to JSON file."""
        data = {
            "model": self.model,
            "summary": {
                "total_queries": len(self.usage_log),
                "total_cost_usd": sum(u.cost_usd for u in self.usage_log),
            },
            "usage_log": [
                {
                    "user_id": u.user_id,
                    "timestamp": u.timestamp.isoformat(),
                    "query": u.query,
                    "input_tokens": u.input_tokens,
                    "output_tokens": u.output_tokens,
                    "cost_usd": u.cost_usd,
                }
                for u in self.usage_log
            ],
        }

        with open(filepath, "w") as f:
            json.dump(data, f, indent=2)

        print(f"Exported to {filepath}")


# ============================================================
# WORKSHOP COST ESTIMATOR
# ============================================================

def estimate_workshop_cost(
    participants: int = 50,
    queries_per_participant: int = 10,
    avg_input_tokens: int = 500,
    avg_output_tokens: int = 500,
    model: str = DEFAULT_MODEL,
) -> Dict:
    """
    Estimate total workshop cost before running.

    Args:
        participants: Number of workshop participants
        queries_per_participant: Average queries per person
        avg_input_tokens: Average input tokens per query
        avg_output_tokens: Average output tokens per query
        model: Model to use for pricing

    Returns:
        Cost estimate dict
    """
    pricing = PRICING.get(model, PRICING[DEFAULT_MODEL])

    total_queries = participants * queries_per_participant
    total_input = total_queries * avg_input_tokens
    total_output = total_queries * avg_output_tokens

    input_cost = (total_input / 1_000_000) * pricing["input_per_1m"]
    output_cost = (total_output / 1_000_000) * pricing["output_per_1m"]
    total_cost = input_cost + output_cost

    return {
        "participants": participants,
        "queries_per_participant": queries_per_participant,
        "total_queries": total_queries,
        "total_input_tokens": total_input,
        "total_output_tokens": total_output,
        "estimated_cost_usd": total_cost,
        "cost_per_participant": total_cost / participants,
        "model": model,
    }


# ============================================================
# DEMO
# ============================================================

if __name__ == "__main__":
    # Estimate workshop cost
    print("Workshop Cost Estimate")
    print("-" * 40)
    estimate = estimate_workshop_cost(
        participants=50,
        queries_per_participant=10,
    )
    for key, value in estimate.items():
        if isinstance(value, float):
            print(f"{key}: ${value:.2f}" if "cost" in key else f"{key}: {value:,.0f}")
        else:
            print(f"{key}: {value}")

    print("\n")

    # Demo tracking
    print("Demo Tracking")
    print("-" * 40)
    tracker = WorkshopCostTracker()

    # Simulate some queries
    tracker.log_tokens_directly("user_1", "Find flights to Tokyo", 450, 320)
    tracker.log_tokens_directly("user_1", "What about hotels?", 380, 450)
    tracker.log_tokens_directly("user_2", "Visa requirements for Japan", 280, 520)
    tracker.log_tokens_directly("user_3", "Book a trip to Paris", 520, 680)

    tracker.print_report()

## State Management (Exercise 4)

The agent remembers user preferences using ADK state prefixes:

### State Prefix Patterns

| Prefix | Scope | Persists? | Example |
|--------|-------|-----------|---------|
| (none) | Session | Until session ends | `state["current_step"] = 3` |
| `user:` | User | Across all sessions | `state["user:budget"] = 1500` |
| `temp:` | Invocation | Never | `state["temp:api_response"] = {...}` |
| `app:` | Application | All users | `state["app:version"] = "1.0"` |

### Using Preferences

**Storing preferences (in tool functions):**
```python
from google.adk.context import ToolContext

def remember_preference(context: ToolContext, pref_type: str, value: str):
    context.state[f"user:{pref_type}"] = value
    return f"I'll remember your {pref_type} is {value}"
```

**Injecting into agent instructions:**
```python
instruction='''You are a travel assistant.

User preferences:
- Budget: {user:budget?}
- Travel style: {user:travel_style?}

Apply preferences when making recommendations.
'''
```

Note the `?` makes keys optional (won't error if not set).

See `state_utils.py` for reusable preference management utilities.
```
