Create comprehensive bash script for RAG corpus setup following Pattern 1 from research.

Script structure:
```bash
#!/bin/bash
set -e

# Configuration
PROJECT_ID="${1:-$GOOGLE_CLOUD_PROJECT}"
LOCATION="${2:-us-central1}"
BUCKET_NAME="${PROJECT_ID}-adk-workshop-rag"
CORPUS_NAME="travel-destination-guides"
GUIDES_DIR="$(dirname "$0")/../destination-guides/pdf"

# Validation
if [ -z "$PROJECT_ID" ]; then
    echo "Usage: $0 <project-id> [location]"
    echo "Or set GOOGLE_CLOUD_PROJECT environment variable"
    exit 1
fi

echo "==================================================="
echo "RAG Corpus Setup for ADK Workshop"
echo "==================================================="
echo "Project: $PROJECT_ID"
echo "Location: $LOCATION"
echo "Bucket: gs://$BUCKET_NAME"
echo ""

# 1. Authenticate check
echo "[1/6] Checking authentication..."
gcloud auth print-access-token > /dev/null || { echo "Please run: gcloud auth login"; exit 1; }

# 2. Create GCS bucket
echo "[2/6] Creating GCS bucket..."
gsutil mb -p "$PROJECT_ID" -l "$LOCATION" "gs://$BUCKET_NAME/" 2>/dev/null || echo "Bucket exists"

# 3. Upload PDFs
echo "[3/6] Uploading destination guides..."
gsutil -m cp "$GUIDES_DIR"/*.pdf "gs://$BUCKET_NAME/guides/"
echo "   Uploaded $(ls "$GUIDES_DIR"/*.pdf | wc -l) PDF files"

# 4. Create RAG corpus
echo "[4/6] Creating RAG corpus..."
CORPUS_RESPONSE=$(curl -s -X POST \
  -H "Authorization: Bearer $(gcloud auth print-access-token)" \
  -H "Content-Type: application/json" \
  "https://${LOCATION}-aiplatform.googleapis.com/v1/projects/${PROJECT_ID}/locations/${LOCATION}/ragCorpora" \
  -d '{
    "display_name": "'"$CORPUS_NAME"'",
    "description": "Destination travel guides for ADK workshop exercises"
  }')

CORPUS_ID=$(echo "$CORPUS_RESPONSE" | jq -r '.name' | awk -F'/' '{print $NF}')
FULL_CORPUS_ID="projects/${PROJECT_ID}/locations/${LOCATION}/ragCorpora/${CORPUS_ID}"
echo "   Corpus ID: $CORPUS_ID"

# 5. Import PDFs with layout parsing
echo "[5/6] Importing PDFs to corpus (this takes 5-10 minutes)..."
curl -s -X POST \
  -H "Authorization: Bearer $(gcloud auth print-access-token)" \
  -H "Content-Type: application/json" \
  "https://${LOCATION}-aiplatform.googleapis.com/v1/${FULL_CORPUS_ID}/ragFiles:import" \
  -d '{
    "import_rag_files_config": {
      "gcs_source": {
        "uris": ["gs://'"$BUCKET_NAME"'/guides/*.pdf"]
      },
      "rag_file_chunking_config": {
        "chunk_size": 1024,
        "chunk_overlap": 256
      },
      "rag_file_parsing_config": {
        "use_advanced_pdf_parsing": true
      },
      "max_embedding_requests_per_min": 1000
    }
  }'

# 6. Output results
echo ""
echo "[6/6] Setup complete!"
echo "==================================================="
echo "CORPUS ID (save this for workshop):"
echo "$FULL_CORPUS_ID"
echo "==================================================="
echo ""
echo "To check indexing status:"
echo "  gcloud ai rag-corpora describe $CORPUS_ID --location=$LOCATION"
echo ""
echo "Save to .env file:"
echo "  echo 'RAG_CORPUS_ID=$FULL_CORPUS_ID' >> .env"
```

#!/usr/bin/env python3
"""
Validate RAG corpus before workshop.

Tests:
1. Corpus exists and is indexed
2. Retrieval returns relevant results
3. Tables/lists are preserved in chunks
4. All destinations are searchable

Usage: python validate-corpus.py [corpus_id]
"""

import os
import sys
from google.adk.tools.retrieval.vertex_ai_rag_retrieval import VertexAiRagRetrieval
from vertexai.preview import rag

# Test queries for each destination
TEST_QUERIES = [

    ("Best time to visit Paris?", "paris", ["spring", "fall"]),
    ("Top attractions in New York City?", "new york", ["Statue of Liberty", "Empire State"]),
    ("Cultural customs in Singapore?", "singapore", ["strict laws", "multi-cultural"]),
    ("Transportation in London?", "london", ["Tube", "Oyster"]),
    ("What is the entry fee for Tokyo Skytree?", "tokyo", ["¥", "2100"]),  # Table content test
    ("Rome food recommendations?", "rome", ["pasta", "carbonara"]),
    ("Bangkok street food prices?", "bangkok", ["THB", "budget"]),
    ("Sydney beach safety?", "sydney", ["rip", "sun"]),
    ("Dubai dress code requirements?", "dubai", ["modest", "Islamic"]),
]

        vector_distance_threshold=0.5,  # Lower threshold for testing
    )

    passed = 0
    failed = 0

    for query, expected_dest, expected_keywords in TEST_QUERIES:
        print(f"Query: {query}")
        # ... test implementation
        # Check if expected keywords appear in retrieved chunks
        pass

    print(f"\n{'='*60}")
    print(f"Results: {passed}/{len(TEST_QUERIES)} passed")
    return failed == 0

def main():
    corpus_id = sys.argv[1] if len(sys.argv) > 1 else os.environ.get('RAG_CORPUS_ID')
    if not corpus_id:
        print("Usage: python validate-corpus.py <corpus_id>")
        print("Or set RAG_CORPUS_ID environment variable")
        sys.exit(1)
