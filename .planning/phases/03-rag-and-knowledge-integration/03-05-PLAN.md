## Why RAG Matters for Travel Assistants

Your agent from Exercise 2 can search flights and hotels using function tools.
But what about questions like:
- "What are the visa requirements for Japan?"
- "What's the best time to visit Paris?"
- "What cultural customs should I know in Tokyo?"

This information doesn't come from real-time APIs - it comes from travel guides,
documentation, and knowledge bases. RAG lets your agent access this static knowledge.

## Tools vs RAG Decision Flowchart

```
User query arrives
        |
        v
Does it need REAL-TIME data?
(prices, availability, status)
        |
      YES --> Use FUNCTION CALLING TOOLS
        |     (search_flights, search_hotels)
        |
       NO
        |
        v
Is it STATIC KNOWLEDGE?
(guides, policies, facts)
        |
      YES --> Use RAG RETRIEVAL
        |     (destination_knowledge)
        |
       NO --> LLM answers directly
```

# Helper function to run agent and show response
async def ask_destination_expert(query: str):
    """Run a query against the destination expert agent."""
    session_service = InMemorySessionService()
    session = await session_service.create_session(
        app_name=destination_expert.name,
        user_id='workshop_user'
    )

    runner = Runner(
        agent=destination_expert,
        session_service=session_service,
        app_name=destination_expert.name
    )

    print(f"You: {query}")
    print("-" * 50)

    async for event in runner.run_async(
        user_id='workshop_user',
        session_id=session.id,
        new_message=Content(parts=[Part(text=query)], role="user")
    ):
        # Show tool calls
        if hasattr(event.content, 'parts'):
            for part in event.content.parts:
                if hasattr(part, 'function_call') and part.function_call:
                    print(f"[RAG Tool Called]")

        if event.is_final_response():
            print(f"Agent: {event.content.parts[0].text}")
            break

# For Colab compatibility
def ask(query: str):
    asyncio.get_event_loop().run_until_complete(ask_destination_expert(query))

# SOLUTION: Complete hybrid assistant

async def hybrid_assistant(query: str) -> str:
    """Coordinate booking and destination agents."""
    query_lower = query.lower()
    session_service = InMemorySessionService()

    async def run_agent(agent, q):
        session = await session_service.create_session(
            app_name=agent.name,
            user_id='hybrid_user'
        )
        runner = Runner(
            agent=agent,
            session_service=session_service,
            app_name=agent.name
        )
        async for event in runner.run_async(
            user_id='hybrid_user',
            session_id=session.id,
            new_message=Content(parts=[Part(text=q)], role="user")
        ):
            if event.is_final_response():
                return event.content.parts[0].text
        return ""

    # Detect booking intent
    needs_booking = any(word in query_lower for word in
                        ['flight', 'hotel', 'book', 'search', 'price'])
    needs_knowledge = any(word in query_lower for word in
                          ['visa', 'weather', 'culture', 'attraction', 'tips'])

    # Extract destination for enrichment
    destinations = {'tokyo': 'Tokyo', 'paris': 'Paris', 'new york': 'New York'}
    destination = None
    for key, city in destinations.items():
        if key in query_lower:
            destination = city
            break

    # Route and optionally enrich
    if needs_booking:
        print("Booking Agent")
        result = await run_agent(booking_agent, query)

        if destination:
            print(f"+ {destination} Tips")
            tips = await run_agent(destination_expert,
                                   f"Top 3 tips for visiting {destination}")
            return f"{result}\n\n**{destination} Travel Tips:**\n{tips}"
        return result

    elif needs_knowledge:
        print("Destination Expert")
        return await run_agent(destination_expert, query)

    else:
        print("Booking Agent (default)")
        return await run_agent(booking_agent, query)

# Test
async def demo():

    response = await hybrid_assistant(
        "Find flights from SFO to Tokyo on March 15 and tell me about visa requirements"
    )
    print(response)

asyncio.get_event_loop().run_until_complete(demo())
